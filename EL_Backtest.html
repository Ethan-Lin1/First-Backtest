def get_stock_data(stock,startdate,enddate,period,interval):
        ticker = stock  
        #overrides Pandas datareader to specify what columns to download from yf
        yf.pdr_override()
        #download specific stock from a specified date range at a specific interval and period
        df = yf.download(tickers=stock, start=startdate, end=enddate, interval=interval,period=period)
        #reset the dataframes index without creating a new dataframe
        df.reset_index(inplace=True) 
        #change the column name
        df['date'] = df['Date'].dt.date
      
        return df

def ema_strategy(df,short_MA,long_MA):
    #calculate the mean ewm for the short_MA, span determines alpha = 2/(span + 1)
    df['short_MA'] = df['Close'].ewm(span = short_MA).mean()
    df['long_MA'] = df['Close'].ewm(span = long_MA).mean()
    #when the short_MA is less than the long_MA then 1, else 0
    df['crosszero'] = np.where(df['short_MA'] < df['long_MA'], 1.0, 0.0)
    #the position taken is the change in crosszero
    df['position'] = df['crosszero'].diff()
    #exit the position at the end
    df['position'].iloc[-1] = -1
    for i, row in df.iterrows():
        # if theres a positive difference in crosszero, then the long_MA must have passed the short MA
        if df.loc[i,'position'] == 1 :
                #thus buy at the close price (rounded to 2 decimals)
                buy_price = round(df.loc[i,'Close'],2)
                df.loc[i,'buy'] = buy_price
        if df.loc[i,'position'] == -1 :
                #opposite of above
                sell_price = round(df.loc[i,'Close'],2)
                df.loc[i,'sell'] = sell_price
    
    return df

def graph(df,stock):
    #plot the graph
    fig = plt.figure(figsize = [15,6])
    ax = plt.subplot(1,1,1)
    #label long_MA blue, short_MA orange, and the close black
    ax.plot(df['Date'], df['long_MA'], color = 'blue', label = 'long MA')
    ax.plot(df['Date'], df['short_MA'], color = 'orange', label = 'short MA')
    ax.plot(df['Date'], df['Close'], color = 'black', label = 'Close')
    #buy sell markers
    ax.plot(df['Date'], df['buy'], color = 'green', label = 'Buy',marker = '^')
    ax.plot(df['Date'], df['sell'], color = 'red', label = 'Sell',marker = 'v')
    ax.legend(loc = 'upper right')
    ax.set_xlabel('Date')
    ax.set_title(stock)
    plt.show()
